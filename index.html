fetch('./src/webstack.json') .then(response => { if (!response.ok) throw new Error('找不到 json 数据文件，请检查路径'); return response.json(); }) .then(data => { // 兼容性处理：如果数据包裹在 results 字段里 const items = Array.isArray(data) ? data : (data.results || []); if (items.length === 0) throw new Error('JSON 数据为空，请检查 Notion 同步是否成功'); const siteStructure = {}; // 第一步：整理数据 items.forEach(item => { // 提取字段，如果 Notion 没填则显示“未分类” const top = (item.category && item.category.name) || item.category || '未分类'; const sub = (item.tags && item.tags.name) || item.tags || '默认'; if (!siteStructure[top]) siteStructure[top] = {}; if (!siteStructure[top][sub]) siteStructure[top][sub] = []; siteStructure[top][sub].push(item); }); // 第二步：渲染左侧菜单 const menuContainer = document.getElementById('main-menu'); if (menuContainer) { menuContainer.innerHTML = Object.keys(siteStructure).map(topName => `
  • ${topName}
    • `).join(''); } // 第三步：渲染右侧内容 const contentContainer = document.getElementById('main-content'); if (contentContainer) { contentContainer.innerHTML = Object.keys(siteStructure).map(topName => `

    ${topName}

    ${Object.keys(siteStructure[topName]).map(subName => `
    ${subName}
    ${siteStructure[topName][subName].map(item => `
    ${item.title || '无标题'}

    ${item.summary || '暂无描述'}

    `).join('')}
    `).join('')} `).join(''); } }) .catch(err => { console.error(err); // 如果报错，在页面上提示你 document.body.insertAdjacentHTML('afterbegin', `
    配置错误提示: ${err.message}
    `); });
